CREATE TABLE WORDS(
	USER_ID varchar(50) NOT NULL,
	WORD varchar(50) NOT NULL,
	PRONOUNCE varchar(50),
	VOCABULARY varchar(50),
	NEXT_TIME int NOT NULL DEFAULT 0,
	STUDY_TIME int NOT NULL DEFAULT 0,
	TIMES int NOT NULL DEFAULT 0,
	FAVORITE char(1) NOT NULL DEFAULT '0',
	SOUND varchar(50)
);

ALTER TABLE WORDS ADD PRIMARY KEY(USER_ID,WORD);

CREATE TABLE TIMES(
	USER_ID varchar(50) NOT NULL,
	TIMES int NOT NULL,
	DAY_DELAY int NOT NULL
);


CREATE TABLE USERS(
	USER_ID varchar(50) NOT NULL,
	PAGE_OFFSET int DEFAULT 0,
	DAY_LIMIT int DEFAULT 0
);

ALTER TABLE USERS ADD PRIMARY KEY(USER_ID);
