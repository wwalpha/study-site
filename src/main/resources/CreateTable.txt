DROP TABLE `words`

CREATE TABLE `words` (
  `USER_ID` varchar(50) NOT NULL,
  `WORD_NO` int NOT NULL,
  `WORD` varchar(50) NOT NULL,
  `PRONOUNCE` varchar(50) DEFAULT NULL,
  `VOCABULARY` varchar(50) DEFAULT NULL,
  `NEXT_TIME` int(11) NOT NULL DEFAULT '0',
  `STUDY_TIME` int(11) NOT NULL DEFAULT '0',
  `TIMES` int(11) NOT NULL DEFAULT '0',
  `FAVORITE` char(1) NOT NULL DEFAULT '0',
  `SOUND` varchar(500) DEFAULT NULL,
  PRIMARY KEY (`USER_ID`,`WORD_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=UTF8;


CREATE TABLE TIMES(
	USER_ID varchar(50) NOT NULL,
	TIMES int NOT NULL,
	DAY_DELAY int NOT NULL
);

CREATE TABLE USERS(
	USER_ID varchar(50) NOT NULL,
	PAGE_OFFSET int DEFAULT 0,
	DAY_LIMIT int DEFAULT 0
