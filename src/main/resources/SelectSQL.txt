-- NEW WORD
SELECT DISTINCT
	WORD,
	PRONOUNCE,
	VOCABULARY,
	NEXT_TIME AS NEXTTIME
	STUDY_TIME AS STUDYTIME
	TIMES,
	FAVORITE,
	SOUND
FROM
    WORDS
WHERE
    USER_ID = %1%
    AND NEXT_TIME <= CAST(DATE_FORMAT(NOW(),'%Y%m%d') AS UNSIGNED)
    AND TIMES = 0
    AND TYPE IN (%2%)
ORDER BY
    NEXT_TIME DESC,
    INDEX
LIMIT 49
